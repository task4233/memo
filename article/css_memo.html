<script>
export default {
  name: '#headerWrap',
  data () {
    return {
      headerMenu_state: false,
      modal_bk: false
    }
  },
  methods: {
    mainMenuToggle: function () {
      if (this.headerMenu_state) {
        this.headerMenu_state = false
        this.modal_bk = false
        return
      }
      this.headerMenu_state = true
      this.modal_bk = true
    },
    modalBkClick: function () {
      this.headerMenu_state = false
      this.modal_bk = false
    }
  }
}
</script>
